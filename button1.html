void move(char* s) {
    int i, l = strlen(s);
    char t = s[0];
    for (i = 0; i < l - 1; i++) {
        s[i] = s[i + 1];
    }
    s[i] = t;
}
void remove2(char* s) {
    int i, l = strlen(s);
    char t = s[l - 1];
    for (i = l; i > 0; i--) {
        s[i] = s[i - 1];
    }
    s[0] = t;
}
